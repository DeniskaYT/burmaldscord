function start() { myNick = document.getElementById('nick').value; if(!myNick) return; document.getElementById('auth-overlay').style.display='none'; document.getElementById('app').style.display='flex'; // Создаем объект Peer myPeer = new Peer(undefined, { host: location.hostname, port: location.port || (location.protocol === 'https:' ? 443 : 80), path: '/peerjs', secure: location.protocol === 'https:' }); myPeer.on('open', id => { console.log("Мой ID получен:", id); // Добавил лог для проверки socket.emit('register-user', { peerId: id, name: myNick }); }); myPeer.on('error', err => { console.error("Ошибка PeerJS:", err.type); if(err.type === 'peer-unavailable') alert("Кент не в сети или ID неверный"); }); myPeer.on('call', c => { currentCall = c; document.getElementById('call-overlay').style.display='flex'; document.getElementById('ring').play().catch(e => console.log("Музло заблочено")); }); } async function call() { const id = document.getElementById('tid').value; if(!id) return alert("Введи ID, ежжи!"); // ВОТ ТУТ ПРОВЕРКА: if (!myPeer || !myPeer.id) { return alert("Погоди, связь еще не установилась (Peer не готов)!"); } try { myStream = await navigator.mediaDevices.getUserMedia({ audio: true }); console.log("Микрофон захвачен, звоню на:", id); const c = myPeer.call(id, myStream); c.on('stream', s => { const a = document.createElement('audio'); a.srcObject = s; a.play(); }); } catch (err) { console.error(err); alert("Микрофон не отвечает или ты запретил доступ."); } }
